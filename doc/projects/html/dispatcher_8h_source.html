30 /// \brief dispatcher implements a publisher/subscriber pattern
31 ///
32 /// \param t_data is the type of the data to be handled. If it is not \p void,
33 /// it must be:
34 /// - default constructible
35 /// - move constructible
36 ///
37 /// \tparam t_log provides log funcionality:
38 /// static void debug(const std::string & p_file, int p_line, const
39 /// t_params&... p_params)
40 /// static void info(const std::string & p_file, int p_line, const t_params&...
41 /// p_params)
42 /// static void warn(const std::string & p_file, int p_line, const t_params&...
43 /// p_params)
44 /// static void error(const std::string & p_file, int p_line, const
45 /// t_params&... p_params)
46 /// static void fatal(const std::string & p_file, int p_line, const
47 /// t_params&... p_params)
48 template <typename t_msg, typename t_log> class dispatcher_t {
49 public:
50  /// \brief work_t is the type of work function, i.e., the function that will
51  /// be called in a loop in order to execute some work
52  ///
53  /// \param t_data is an instance of the data to be handled
54  ///
55  /// \return result::stop if the loop where this function is being
56  /// called should stop, or work_status::dont_stop if it should continue
57  typedef typename traits_t<t_msg>::worker worker;

61  ///

66  ///

85  ///
86  /// Stops all the \p thread_pool
87  ~dispatcher_t() {
88  concurrent_log_debug(log, "leaving");
89  typename thread_pool_list::iterator _end = m_thread_pool_list.end();

91  _ite != _end; ++_ite) {

96  ///
97  /// \brief adds a group of \p work_t objects to handle \p t_msg objects
98  /// \param p_work_factory is a function that creates \p work_t objects
99  /// \param p_num_handlers is the number of \p work_t objects that we want to
100  /// "fight" among each other to handle a \p t_msg object
101  /// \param p_work_timeout is the maximum amount of time for a \p work_t
102  /// function can take to handle a \p t_msg instance
103  static void subscribe(const queue &p_queue,
104  std::function<worker()> p_work_factory,
105  uint16_t p_num_handlers,
106  std::chrono::milliseconds p_work_timeout) {
107 
108  // creating a \p thread_pool
109  thread_pool _thread_pool;
110 
111  // adding the \p work_t functions
112  _thread_pool.add_work(p_num_handlers, p_work_factory, p_work_timeout);
113 
114  // running the \p thread_pool
115  _thread_pool.run();
116 
117  // adding the \p thread_pool to the list
118  // m_thread_pool_list.push_back(std::move(_thread_pool));
119  m_thread_pool_list.insert({p_queue, std::move(_thread_pool)});
120  }

145  ///
146  /// \brief handle sends a message to the \p work_t objects to be handled
147  /// \param p_msg an instance of \p t_msg
148  ///
149  static void publish(const t_msg &p_msg) {
150  if (m_thread_pool_list.empty()) {
151  concurrent_log_warn(log, "there are no subscribers for ", p_msg);
152  }
153  concurrent_log_debug(log, "# of pools: ", m_thread_pool_list.size());
154  // for (thread_pool& _pool : m_thread_pool_list) {
155  // _pool.handle(p_msg);
156  // }
157  typename thread_pool_list::iterator _end = m_thread_pool_list.end();
158  for (typename thread_pool_list::iterator _ite = m_thread_pool_list.begin();
159  _ite != _end; ++_ite) {
160  _ite->second.handle(p_msg);
161  }
162  }
163 
164  ///
165  /// \brief handle sends a message to the \p work_t objects to be handled
166  /// \tparam t_params are the types of parameters to create a \p t_msg object
167  /// \param p_params are theparameters values to create a \p t_msg object
168  ///
169  template <typename... t_params>
170  inline static void publish(t_params... p_params) {
171  const t_msg _msg(p_params...);
172  publish(_msg);
173  }
174 
175  static void stop() {
176  typename thread_pool_list::iterator _end = m_thread_pool_list.end();
177  for (typename thread_pool_list::iterator _ite = m_thread_pool_list.begin();
178  _ite != _end; ++_ite) {
179  _ite->second.stop();
180  }
181  }
