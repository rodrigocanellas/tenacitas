22 /// \brief Formats a number with the maximum characters necessary to represent
23 /// the number to string, filled with \p p_fill and aligned with \p p_align
24 ///
25 /// \tparam t_num_type is the type of the number to be formated
26 ///
27 /// \param p_num is the value of the number to be formated
28 ///
29 /// \param p_fill is the char used to fill the parts of the string not filled
30 /// with \p p_num digits
31 ///
32 /// \param p_align defines if \p p_num will be aligned at left, right, or center
33 /// of the string
34 ///
35 template <typename t_num_type>
36 inline std::string format_000(t_num_type p_num, char p_fill = '0',
37  align p_align = align::right) {
38  std::stringstream _stream;
39  _stream << (p_align == align::right ? std::right : std::left)
40  << std::setfill(p_fill) << std::setw(max_str_length<t_num_type>())
41  << p_num;
42  return _stream.str();
43 }
