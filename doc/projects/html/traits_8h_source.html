27 /// \brief loop_traits defines types when dealing with loop with data
28 ///
29 /// \param t_data is the type of the data to be handled; it must be:
30 /// - default constructible
31 /// - move constructible
32 ///
33 template<typename t_data>
34 struct traits_t
35 {
36  ///
37  /// \brief work_t is the type of work function, i.e., the function that will
38  /// be called in a loop in order to execute some work
39  ///
40  /// \param t_data is an instance of the data to be handled
41  ///
42  /// \return result::stop if the loop where this function is being
43  /// called should stop, or work_status::dont_stop if it should continue
44  typedef std::function<work_status(t_data&&)> worker;
45 
46  typedef std::shared_ptr<worker> worker_ptr;
47 
48  ///
49  /// \brief provide_t is the type of function that provides data to the work
50  /// function during the loop execution
51  ///
52  /// \return a pair, where if \p first is \p true, the \p second has a
53  /// meaningful data; if \p first is \p false, then \p second has a default
54  /// value of \p t_data
55  ///
56  typedef std::function<std::pair<bool, t_data>()> provider;
57 
58  ///
59  /// \brief break_t is the type of function that indicates if the loop should
60  /// stop
61  ///
62  /// \return result::stop if the loop where this function is being
63  /// called should stop, or work_status::dont_stop if it should continue
64  typedef std::function<work_status()> breaker;
65 };
