27 /// \brief sleeping_loop allows a function (object) to be called in a loop that
28 /// is executed asyncronously at a user defined period of time.
29 ///
30 /// \param t_data is the type of the data to be handled. If it is not \p void,
31 /// it must be:
32 /// - default constructible
33 /// - move constructible
34 ///
35 /// \tparam t_log provides log funcionality:
36 /// static void debug(const std::string & p_file, int p_line, const
37 /// t_params&... p_params)
38 /// static void info(const std::string & p_file, int p_line, const t_params&...
39 /// p_params)
40 /// static void warn(const std::string & p_file, int p_line, const t_params&...
41 /// p_params)
42 /// static void error(const std::string & p_file, int p_line, const
43 /// t_params&... p_params)
44 /// static void fatal(const std::string & p_file, int p_line, const
45 /// t_params&... p_params)
46 template<typename t_data, typename t_log>
47 struct sleeping_loop_t
48 {
49  /// \brief work_t is the type of work function, i.e., the function that will
50  /// be called in a loop in order to execute some work
51  ///
52  /// \param t_data is an instance of the data to be handled
53  ///
54  /// \return result::stop if the loop where this function is being
55  /// called should stop, or work_status::dont_stop if it should continue
56  typedef typename traits_t<t_data>::worker worker;
57 
58  ///
59  /// \brief provide_t is the type of function that provides data to the work
60  /// function during the loop execution
61  ///
62  /// \return a pair, where if \p first is \p true, the \p second has a
63  /// meaningful data; if \p first is \p false, then \p second has a default
64  /// value of \p t_data
65  ///
66  typedef typename traits_t<t_data>::provider provider;
67 
68  ///
69  /// \brief log alias for @p t_log
70  ///
71  typedef t_log log;
72 
73  ///
74  /// \brief sleeping_loop creates a \p sleeping_loop object, when <tt>bt_data
75  /// != void</tt>, and a \p provider is necessary
76  ///
77  /// \param p_interval the amount of time that the loop will sleep between
78  /// executions
79  /// \param p_work function that will be executed each time the loop wakes up
80  /// \param p_timeout amount of time that the loop will wait for \p p_work to
81  /// execute
82  /// \param p_provide function that will provide data to the work function,
83  /// each time the loop wakes up
84  ///
85  sleeping_loop_t(std::chrono::milliseconds p_interval,
86  worker&& p_work,
87  std::chrono::milliseconds p_timeout,
88  provider&& p_provide)
89  : m_async(std::move(p_work),
90  std::move(p_timeout),
91  [this]() -> work_status { return this->break_loop(); },
92  std::move(p_provide))
93  , m_interval(p_interval)
94  {}
95 
96  ///
97  /// \brief sleeping_loop creates a \p sleeping_loop object, when <tt>bt_data
98  /// == void</tt>, and a \p provider is not necessary
99  ///
100  /// \param p_interval the amount of time that the loop will sleep between
101  /// executions
102  /// \param p_work function that will be executed each time the loop wakes up
103  /// \param p_timeout amount of time that the loop will wait for \p p_work to
104  /// execute
105  ///
106  sleeping_loop_t(std::chrono::milliseconds p_interval,
107  worker p_work,
108  std::chrono::milliseconds p_timeout)
109  : m_async(p_work,
110  p_timeout,
111  [this]() -> work_status { return this->break_loop(); },
112  []() -> void {})
113  , m_interval(p_interval)
114  {}

158  /// \return a copy of the function that provides an instance of \p t_data,

160  ///
161  inline provider get_provide() const { return m_async.get_provide(); }

165  /// \return the amount of time that the loop will wait for the work function

171  }

174  /// \brief get_interval
175  /// \return the interval of loop execution
176  ///
177  inline std::chrono::milliseconds get_interval() const { return m_interval; }
178 
179  /// \brief run starts the loop
180  void run()
181  {
182  if (!m_async.is_stopped()) {
183  concurrent_log_debug(
184  log, this, " not running async loop because it was not stopped");
185  return;
186  }
187  concurrent_log_debug(log, this, " running async loop");
188  m_async.run();
189  }
190 
191  /// \brief stop stops the loop
192  void stop()
193  {
194  if (m_async.is_stopped()) {
195  concurrent_log_debug(
196  log, this, " not stopping async loop because it was not running");
197  return;
198  }
199  concurrent_log_debug(log, this, " stop");
200  m_cond_var.notify_all();
201 
202  concurrent_log_debug(log, this, " all notified, and m_async = ", &m_async);
203  m_async.stop();
204  }

207  /// \brief a simpler name
208  typedef async_loop_t<t_data, t_log> async_loop;
209 
210 private:
211  ///
212  /// \brief move a \p sleeping_loop to this
213  /// \param p_sleep the \p sleeping_loop to be moved
214  ///
215  inline void move(sleeping_loop_t&& p_sleep) noexcept
216  {
217  // save if the right side was running
218  bool _stopped = p_sleep.is_stopped();
219  // stop the right side
220  p_sleep.stop();
221 
222  // move the interval
223  m_interval = std::move(p_sleep.m_interval);
224 
225  // move the async_loop, reseting the break loop
226  m_async = async_loop(std::move(p_sleep.get_work()),
227  std::move(p_sleep.get_timeout()),
228  [this]() -> work_status { return this->break_loop(); },
229  std::move(p_sleep.get_provide()));
230 
231  // if the right side was not stopped
232  if (!_stopped) {
233  // run this sleeping_loop
234  run();
235  }
236  }
237 
238  ///
239  /// \brief break_loop function that defines if the loop should stop
240  /// \return \p true if the loop should break; \p false othewise
241  ///
242  work_status break_loop()
243  {
244  std::unique_lock<std::mutex> _lock(m_mutex);
245  if (m_cond_var.wait_for(_lock, m_interval) == std::cv_status::timeout) {
246  // timeout, so do not stop
247  concurrent_log_debug(log, this, " must not stop");
248  return work_status::dont_stop;
249  }
250  // no timeout, so do stop
251  concurrent_log_debug(log, this, " must stop");
252  return work_status::stop;
253  }
254 
255 private:
256  /// \brief m_loop the async_loop used
257  async_loop m_async;
258 
259  /// \brief m_interval

261 
262  /// \brief m_mutex
263  std::mutex m_mutex;
264 
265  /// \brief m_cond_var
266  std::condition_variable m_cond_var;
267 };
