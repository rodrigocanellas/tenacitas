40 /// \param t_data is the type of the data to be handled. If it is not \p void,

44 ///
45 /// \tparam t_log provides log funcionality:
46 /// static void debug(const std::string & p_file, int p_line, const
47 /// t_params&... p_params)
48 /// static void info(const std::string & p_file, int p_line, const t_params&...

52 /// static void error(const std::string & p_file, int p_line, const
53 /// t_params&... p_params)
54 /// static void fatal(const std::string & p_file, int p_line, const
55 /// t_params&... p_params)
56 template <typename t_data, typename t_log> struct async_loop_t {
57  /// \brief work_t is the type of work function, i.e., the function that will
58  /// be called in a loop in order to execute some work
59  ///
60  /// \param t_data is an instance of the data to be handled
61  ///
62  /// \return result::stop if the loop where this function is being
63  /// called should stop, or work_status::dont_stop if it should continue
64  typedef typename traits_t<t_data>::worker worker;
65 
66  typedef typename traits_t<t_data>::worker_ptr worker_ptr;
67 
68  ///
69  /// \brief provide_t is the type of function that provides data to the work
70  /// function during the loop execution
71  ///
72  /// \return a pair, where if \p first is \p true, the \p second has a
73  /// meaningful data; if \p first is \p false, then \p second has a default
74  /// value of \p t_data
75  ///
76  typedef typename traits_t<t_data>::provider provider;
77 
78  ///
79  /// \brief break_t is the type of function that indicates if the loop should
80  /// stop
81  ///
82  /// \return result::stop if the loop where this function is being
83  /// called should stop, or work_status::dont_stop if it should continue
84  typedef std::function<work_status()> breaker;
85 
86  ///
87  /// \brief log alias for @p t_log
88  ///
89  typedef t_log log;
90 
91  ///
92  /// \brief async_loop constructor
93  ///
94  /// This constructor should be used when \p t_data is not \p void
95  ///
96  /// \param p_work instance of the function that will execute the defined
97  /// work
98  /// \param p_timeout amount of time that the loop will wait for \p p_work to
99  /// execute
100  /// \param p_break instance of the function that will indicate when the loop
101  /// must stop
102  /// \param p_provide instance of the function that will provide an instance
103  /// of \p t_data, if available
104  ///
105  inline async_loop_t(worker p_work, std::chrono::milliseconds p_timeout,
106  breaker p_break, provider p_provide)
107  : m_loop(p_work, p_timeout, p_break, p_provide), m_thread() {}
108 
109  ///
110  /// \brief async_loop constructor
111  ///
112  /// This constructor should be used when \p t_data is \p void
113  ///
114  /// \param p_work instance of the function that will execute the defined
115  /// work
116  /// \param p_timeout amount of time that the loop will wait for \p p_work to
117  /// execute
118  /// \param p_break instance of the function that will indicate when the loop
119  /// must stop
120  ///
121  inline async_loop_t(worker p_work, std::chrono::milliseconds p_timeout,
122  breaker p_break)
123  : m_loop(p_work, p_timeout, p_break, []() -> void {}), m_thread() {}

187  ///
188  /// \brief run starts the loop
189  ///
190  void run() {
191 
192  if (!m_loop.is_stopped()) {
193  concurrent_log_debug(
194  log, this, " not starting the loop because it is already running");
195  return;
196  }
197  concurrent_log_debug(log, this, " starting the loop");
198  run_core();
199  }
200 
201  ///
202  /// \brief stop stops the loop
203  ///
204  void stop() {
205  if (m_loop.is_stopped()) {
206  concurrent_log_debug(log, this,
207  " not stopping the loop because it was not running");
208  return;
209  }
210 
211  std::lock_guard<std::mutex> _lock(m_mutex);
212  concurrent_log_debug(log, this, " marking to stop");
213  m_loop.stop();
214  concurrent_log_debug(log, this, " joining");
215  }

219  typedef loop_t<t_data, t_log> loop;
220 
221  /// \brief thread_t is an easier name for our wrapper to std::thread
222  typedef thread thread_t;
223 
224 private:
225  /// \brief common method to start the loop
226  void run_core() {
227  std::lock_guard<std::mutex> _lock(m_mutex);
228  m_thread = thread([this]() -> void { m_loop.start(); });
229  }
230

232  /// \brief m_loop is the \p loop to be executed asyncronously
233  loop m_loop;
234
