23 /// \brief loop implements a generic loop
24 ///
25 /// A loop needs a work function, that will execute a defined work at each round
26 /// of the loop; an amount of time that the work function has to execute; a
27 /// break function, that indicates when the loop should break; and a provider
28 /// function, that will provide data for the work function, if available
29 ///
30 /// In this version of the loop class, the work function actually needs a data,
31 /// provided by the provide function, which type is defined by the type
32 /// parameter \p t_data.
33 ///
34 /// \tparam t_data is the type of the data manipulated during the loop; it must
35 /// be:
36 /// - default constructible
37 /// - move constructible
38 ///
39 /// \tparam t_log provides log funcionality:
40 /// static void debug(const std::string & p_file, int p_line, const
41 /// t_params&... p_params)
42 /// static void info(const std::string & p_file, int p_line, const t_params&...
43 /// p_params)
44 /// static void warn(const std::string & p_file, int p_line, const t_params&...
45 /// p_params)
46 /// static void error(const std::string & p_file, int p_line, const
47 /// t_params&... p_params)
48 /// static void fatal(const std::string & p_file, int p_line, const
49 /// t_params&... p_params)
50 template <typename t_data, typename t_log> struct loop_t {
51

56  ///
57  /// \return result::stop if the loop where this function is being
58  /// called should stop, or work_status::dont_stop if it should continue
59  typedef typename traits_t<t_data>::worker worker;

66  /// meaningful data; if \p first is \p false, then \p second has a default

68  ///

75  /// \return result::stop if the loop where this function is being
76  /// called should stop, or work_status::dont_stop if it should continue
77  typedef typename traits_t<t_data>::breaker breaker;

172  ///
173  /// \brief start initates the loop,
174  ///
175  /// If the loop was already initiated, it does not start a new loop
176  ///
177  void start() {
178  if (!m_stopped) {
179  concurrent_log_debug(log, this,
180  " not starting because it was not stopped");
181  return;
182  }
183 
184  m_stopped = false;
185 
186  while (true) {
187 
188  if (m_stopped) {
189  concurrent_log_debug(log, this, " stopping loop");
190  break;
191  }
192 
193  concurrent_log_debug(log, this, " waiting for data");
194  std::pair<bool, t_data> _provide(m_provide());
195  if (!_provide.first) {
196  concurrent_log_debug(log, this, " breaking because there is no data");
197  break;
198  }
199 
200  concurrent_log_debug(log, this, " data received ", _provide.second);
201 
202  std::future<work_status> _future = std::async(
203  std::launch::async,
204  [this](t_data &&p_data) { return this->m_work(std::move(p_data)); },
205  std::move(_provide.second));
206  if (_future.wait_for(m_timeout) == std::future_status::ready) {
207  if (_future.get() == work_status::stop) {
208  concurrent_log_debug(log, this, " worker returned 'result::stop'");
209  m_stopped = true;
210  break;
211  }
212  concurrent_log_debug(log, this,
213  " worker returned 'work_status::dont_stop'");
214  } else {
215  concurrent_log_warn(log, this, " timeout for data ", _provide.second);
216  }
217 
218  if (m_stopped) {
219  concurrent_log_debug(log, this, " stopping loop");
220  break;
221  }
222 
223  if (m_break() == work_status::stop) {
224  concurrent_log_debug(log, this, " stopping because breaker said so");
225  m_stopped = true;
226  break;
227  }
228  }
229  }
