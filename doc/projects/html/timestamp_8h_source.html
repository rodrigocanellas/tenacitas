255  template <typename t_amount> void add(timestamp *p_ts, t_amount p_amount) {
256  days _days_1(p_amount);
257  auto _int = _days_1.integer();
258  auto _dec = _days_1.decimal();
259 
260  seconds _secs_in_day((days(_dec)));
261 
262  seconds _secs(p_ts->m_secs + _secs_in_day);
263 
264  if (days(_secs) > days(1)) {
265  p_ts->m_days += days(_int + 1);
266  p_ts->m_secs = _secs - seconds(days(1));
267  } else {
268  p_ts->m_days += days(_int);
269  p_ts->m_secs += _secs_in_day.integer();
270  }
271  }

273  template <typename t_amount>
274  void substract(timestamp *p_ts, t_amount p_amount) {
275  days _days_1(p_amount);
276  auto _int = _days_1.integer();
277  auto _dec = _days_1.decimal();
278 
279  seconds _secs_in_day((days(_dec)));
280 
281  seconds _secs(p_ts->m_secs - _secs_in_day);
282 
283  if (days(_secs) < 0) {
284  p_ts->m_days -= days(_int + 1);
285  p_ts->m_secs = seconds(days(1)) + _secs;
286  } else {
287  p_ts->m_days -= days(_int);
288  p_ts->m_secs += _secs_in_day.integer();
289  }
290  }
