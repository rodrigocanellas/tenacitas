50  void timestamp::days_secs2dMywhms() {
51  int64_t _days(m_days.value());
52  _days += 719468;
53  const int64_t era = (_days >= 0 ? _days : _days - 146096) / 146097;
54  const auto doe = static_cast<unsigned>(_days - era * 146097); // [0, 146096]
55  const auto yoe =
56  (doe - doe / 1460 + doe / 36524 - doe / 146096) / 365; // [0, 399]
57  auto y = static_cast<int32_t>(yoe) + era * 400;
58  const auto doy = doe - (365 * yoe + yoe / 4 - yoe / 100); // [0, 365]
59  const auto mp = (5 * doy + 2) / 153; // [0, 11]
60  const auto d = doy - (153 * mp + 2) / 5 + 1; // [1, 31]
61  const auto m = mp + (mp < 10 ? 3 : -9); // [1, 12]
62  m_year = year(y + (m <= 2));
63  m_month = month(m);
64  m_day = day(d);
65  m_weekday = weekday(_days >= -4 ? (_days + 4) % 7 : (_days + 5) % 7 + 6);
66  m_hour = hours(m_secs);
67  m_minute = minutes(m_secs - seconds((hours(m_hour))));
68  m_second = seconds(m_secs % 60);
69  m_needs_update = false;
70 }
