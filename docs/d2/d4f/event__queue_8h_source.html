371  }

373  auto _subscriber_id{&p_event_handler};
374 
375  if (m_stopped) {

378 
379  return;
380  }
381 
382  trace(__LINE__, "entering loop");
383 
384  /*
385 template< class Rep, class Period, class Predicate >
386 
387 bool wait_for( std::unique_lock<std::mutex>& lock,
388  const std::chrono::duration<Rep, Period>& rel_time,

390  */

392  while (true) {

395  trace(__LINE__, "locking", _loop_id);
396 
397  std::unique_lock<std::mutex> _lock(m_mutex);
398  m_cond.wait(_lock, [this, _subscriber_id, _loop_id, _queue_id]() -> bool {
399  trace(__LINE__, "entering condition", _loop_id);
400  if (m_paused) {

402  return false;
403  }
404  if (m_stopped) {

406  return true;
407  }
408  if (!m_circular_queue.empty()) {
409  trace(__LINE__, "there is data", _loop_id);
410  return true;

414  });
415  }

417  trace(__LINE__, "lock released", _loop_id);
