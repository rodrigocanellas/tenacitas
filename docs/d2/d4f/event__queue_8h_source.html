323 template <tlc::event t_event>
324 void event_queue<t_event>::subscriber_loop(
325  tld::event_handler<t_event> p_event_handler) {
326  auto _queue_id{m_circular_queue.get_id()};
327  auto _loop_id{std::this_thread::get_id()};
328 
329  trace(__LINE__, "starting subscriber's loop", _loop_id);
330 
331  if (!p_event_handler) {
332  std::stringstream _stream;
333  _stream << "event " << typeid(t_event).name() << "<< queue {" << this
334  << "<<" << m_id << "}<< circular " << m_circular_queue.get_id()
335  << "<< queued_data = " << m_queued_data << "<< tld::event_priority "
336  << m_event_priority << "<< loops = " << &m_loops
337  << "<< stopped = " << m_stopped.load() << "<< thread "
338  << std::this_thread::get_id() << " - "
339  << "invalid event subscriber";
340  const std::string _str{_stream.str()};
341 #ifdef TENACITAS_LOG
342  TNCT_LOG_FAT(_str);
343 #endif
344  throw std::runtime_error(_str);
345  }
346 
347  auto _subscriber_id{&p_event_handler};
348 
349  if (m_stopped) {
350  trace(__LINE__, "leaving because it was already stopped", _loop_id);
351  return;
352  }
353 
354  trace(__LINE__, "entering loop");
355 
356  while (true) {
357  {
358  trace(__LINE__, "locking", _loop_id);
359 
360  std::unique_lock<std::mutex> _lock(m_mutex);
361  m_cond.wait(_lock, [this, _subscriber_id, _loop_id, _queue_id]() -> bool {
362  trace(__LINE__, "entering condition", _loop_id);
363  if (m_paused) {
364  trace(__LINE__, "stopped", _loop_id);
365  return false;
366  }
367  if (m_stopped) {
368  trace(__LINE__, "stopped", _loop_id);
369  return true;
370  }
371  if (!m_circular_queue.empty()) {
372  trace(__LINE__, "there is data", _loop_id);
373  return true;
374  }
375  trace(__LINE__, "waiting", _loop_id);
376  return false;
377  });
378  }
379 
380  trace(__LINE__, "lock released", _loop_id);
381 
382  if (m_stopped) {
383  trace(__LINE__, "stopped due to explicit stop", _loop_id);
384  break;
385  }
386 
387  trace(__LINE__, "about to get the event from the queue", _loop_id);
388 
389  std::optional<t_event> _maybe{m_circular_queue.get()};
390  if (!_maybe.has_value()) {
391  trace(__LINE__, "no event in queue", _loop_id);
392  continue;
393  }
394  t_event _event{std::move(*_maybe)};
395 
396  trace(__LINE__, "about to call subscriber", _loop_id);
397  p_event_handler(std::move(_event));
398 
399  trace(__LINE__, "event handled", _loop_id);
400  }
401  trace(__LINE__, "leaving subscriber's loop", _loop_id);
402 }
